Apps Script -koodi

Kopioi tämä koodi kokonaisuudessaan Apps Script -editoriin (kohta 3 yllä):

const SHEET_NAME = 'Kirjat';
const HEADERS = ['id','title','author','status','date','extra','notes','added'];

function doGet(e) {
  const action = e.parameter.action || '';
  let result;
  try {
    if (action === 'getBooks')        result = getBooks();
    else if (action === 'addBook')    result = addBook(e.parameter);
    else if (action === 'updateBook') result = updateBook(e.parameter);
    else if (action === 'deleteBook') result = deleteBook(e.parameter.id);
    else result = { error: 'Tuntematon action: ' + action };
  } catch(err) { result = { error: err.toString() }; }
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function getSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(HEADERS);
    sheet.getRange(1,1,1,HEADERS.length).setFontWeight('bold').setBackground('#f3f3f3');
    sheet.setFrozenRows(1);
  } else { migrateSheet(sheet); }
  return sheet;
}

function migrateSheet(sheet) {
  const data = sheet.getDataRange().getValues();
  if (!data.length) return;
  const existingHeaders = data[0].map(h => String(h).trim().toLowerCase());
  let needsMigration = false;
  for (var i = 0; i < HEADERS.length; i++) {
    if (existingHeaders[i] !== HEADERS[i]) { needsMigration = true; break; }
  }
  if (!needsMigration && existingHeaders.length >= HEADERS.length) return;
  const newData = [HEADERS];
  for (var r = 1; r < data.length; r++) {
    var oldRow = data[r]; if (!oldRow[0]) continue;
    var o = {};
    existingHeaders.forEach((h,i) => { o[h] = cellToString(oldRow[i]); });
    newData.push([o.id||'',o.title||'',o.author||'',o.status||'',
      o.date||'',o.extra||'',o.notes||'',o.added||'']);
  }
  sheet.clearContents();
  sheet.getRange(1,1,newData.length,HEADERS.length).setValues(newData);
  sheet.getRange(1,1,1,HEADERS.length).setFontWeight('bold').setBackground('#f3f3f3');
  sheet.setFrozenRows(1);
}

function cellToString(value) {
  if (value === null || value === undefined || value === '') return '';
  if (value instanceof Date) {
    const y = value.getFullYear();
    const m = String(value.getMonth()+1).padStart(2,'0');
    const d = String(value.getDate()).padStart(2,'0');
    return y+'-'+m+'-'+d;
  }
  return String(value);
}

function getBooks() {
  const sheet = getSheet();
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return { books: [], success: true };
  const headers = data[0];
  const books = data.slice(1)
    .filter(row => row[0])
    .map(row => { const obj = {}; headers.forEach((h,i) => obj[h]=cellToString(row[i])); return obj; })
    .reverse();
  return { books, success: true };
}

function addBook(params) {
  const sheet = getSheet();
  const data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] == params.id) return { success: false, error: 'Duplikaatti id' };
  }
  sheet.appendRow([params.id||'',params.title||'',params.author||'',params.status||'',
    params.date||'',params.extra||'',params.notes||'',params.added||new Date().toISOString()]);
  return { success: true };
}

function updateBook(params) {
  const sheet = getSheet();
  const data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] == params.id) {
      sheet.getRange(i+1,1,1,HEADERS.length).setValues([[
        params.id||cellToString(data[i][0]),
        params.title||cellToString(data[i][1]),
        params.author||cellToString(data[i][2]),
        params.status||cellToString(data[i][3]),
        params.date||cellToString(data[i][4]),
        params.extra!==undefined?params.extra:cellToString(data[i][5]),
        params.notes!==undefined?params.notes:cellToString(data[i][6]),
        params.added||cellToString(data[i][7])
      ]]);
      return { success: true };
    }
  }
  return addBook(params);
}

function deleteBook(id) {
  const sheet = getSheet();
  const data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    if (data[i][0] == id) { sheet.deleteRow(i+1); return { success: true }; }
  }
  return { success: false, error: 'Kirjaa ei löydy.' };
}

Kirjat eivät synkronoidu toiselle laitteelle

Varmista että molemmilla laitteilla on sama Web App URL syötettynä. Klikkaa ↻ Synkronoi nyt pakottaaksesi synkronoinnin. Jos yhteysindikaattori näyttää punaista, tarkista että Apps Script on julkaistu oikein (Pääsy: Kaikki).

Päivämäärä näyttää väärää tai tyhjenee synkronoinnin jälkeen

Google Sheets muuntaa päivämäärät automaattisesti omaan muotoonsa. Sovellus korjaa tämän automaattisesti — varmista että käytät uusinta Apps Script -koodia.

Sivumäärä tai kuunteluaika menee muistiinpanot-kenttään

Tämä johtuu vanhasta taulukkorakenteesta jossa extra-sarake puuttuu. Uusi Apps Script -koodi korjaa rakenteen automaattisesti ensimmäisellä käyttökerralla.

Yhteys epäonnistuu

Tarkista: 1) URL on oikein kopioitu · 2) Apps Script on julkaistu: Pääsy = Kaikki · 3) Uuden koodiversion jälkeen on julkaistu uutena versiona.

Napit eivät toimi

Tämä tarkoittaa JavaScript-syntaksivirhettä. Lataa uusin versio tiedostosta.
